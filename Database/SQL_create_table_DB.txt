# Entities
Create table IF NOT EXISTS Semester (
Scode Varchar(10) primary key
Foreign key (AYcode) references academic_year (AYcode)
) (DONE)

Create table IF NOT EXISTS class (
Ccode Varchar(10) primary key,
Size Int,
Scode Varchar(10),
Mcode Varchar(10),
Foreign key (SCode) references semester (Scode),
Foreign key (MCode) references module (Mcode)
) (DONE)

Create table IF NOT EXISTS Lecturer (
Lcode Varchar(10) primary key,
Lname Varchar(50)
) (DONE)

Create table IF NOT EXISTS Module (
Mcode Varchar(10) primary key,
Mname Varchar(50),
Pcode Varchar(10),
Foreign key (Pcode) references program (Pcode)
) (DONE)

Create table IF NOT EXISTS Program (
Pcode Varchar(10) primary key,
Pname Varchar(50),
Fcode Varchar(10),
Foreign key (Fcode) references faculty (Fcode)
) (DONE)

Create table IF NOT EXISTS academic_year (
AYcode Varchar(10) primary key
) (DONE)

Create table IF NOT EXISTS Faculty (
Fcode Varchar(10) primary key,
Fname Varchar(50),
AYcode Varchar(10),
Foreign key (AYcode) references academic_year (AYcode)
) (DONE)

Create table IF NOT EXISTS questionaire(
Ccode Varchar(10),
Lcode Varchar(10),
primary key (Ccode, Lcode),
Foreign key (Ccode) references class (Ccode),
Foreign key (Lcode) references lecturer (Lcode)
) (DONE)

# RELATIONSHIPS
Create table IF NOT EXISTS module_program_in_AY(
Pcode Varchar(10),
Mcode Varchar(10),
AYcode Varchar(10),
primary key(Pcode, Mcode, AYcode),
foreign key (Pcode) references Program (Pcode),
foreign key (Mcode) references Module (Mcode),
foreign key (AYcode) references academic_year (AYcode)
) (DONE)

Create table IF NOT EXISTS class_has_lecturer (
Ccode varchar(10),
Lcode varchar(10),
primary code (Ccode, Lcode),
Foreign key (Ccode) references class (Ccode),
Foreign key (Lcode) references lecturer (Lcode)
)
